create table if not exists ponds ( id bigint generated by default as identity primary key, name text, region text, price text, fish text, conditions text, contact text, reviews text, created_at timestamp with time zone default timezone('utc'::text, now()) ); create policy "Allow read ponds" on ponds for select using (true); create policy "Allow insert ponds" on ponds for insert with check (true); create policy "Allow update ponds" on ponds for update using (true); create policy "Allow delete ponds" on ponds for delete using (true);
